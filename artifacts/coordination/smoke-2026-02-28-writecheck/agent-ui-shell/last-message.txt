// tasks-ui/pages/tasks/index.tsx
import Link from "next/link";

type Task = {
  id: string;
  title: string;
  owner: string;
  dueDate: string;
  status: "open" | "in_progress" | "done";
};

const tasks: Task[] = [
  {
    id: "task-1",
    title: "Prepare onboarding checklist",
    owner: "Design",
    dueDate: "Feb 29, 2026",
    status: "open",
  },
  {
    id: "task-2",
    title: "Add UI route entry points",
    owner: "Frontend",
    dueDate: "Mar 02, 2026",
    status: "in_progress",
  },
  {
    id: "task-3",
    title: "Wire task completion endpoint",
    owner: "Backend",
    dueDate: "Mar 04, 2026",
    status: "done",
  },
];

export default function TasksListPage() {
  return (
    <section className="page-card">
      <h1>Task List</h1>
      <ul className="task-list">
        {tasks.map((task) => (
          <li key={task.id} className="task-item">
            <div>
              <strong>{task.title}</strong>
              <div>
                {task.owner} · due {task.dueDate}
              </div>
            </div>
            <span className="badge">{task.status}</span>
            <div style={{ gridColumn: "1 / -1", marginTop: "0.25rem" }}>
              <Link href={`/tasks/${task.id}`} className="button">
                Open task
              </Link>
            </div>
          </li>
        ))}
      </ul>
    </section>
  );
}
