`frontend/src/components/TaskInput.tsx`
```tsx
import { FormEvent, useState } from "react";
import { Task } from "../types/task";

type TaskInputProps = {
  onAddTask: (title: string, notes: string) => void;
};

export const TaskInput = ({ onAddTask }: TaskInputProps) => {
  const [title, setTitle] = useState("");
  const [notes, setNotes] = useState("");

  const handleSubmit = (event: FormEvent) => {
    event.preventDefault();
    const normalizedTitle = title.trim();
    if (!normalizedTitle) {
      return;
    }

    onAddTask(normalizedTitle, notes.trim());
    setTitle("");
    setNotes("");
  };

  return (
    <form onSubmit={handleSubmit} className="task-input">
      <label>
        Task title
        <input
          value={title}
          onChange={(event) => setTitle(event.target.value)}
          placeholder="Add a task..."
        />
      </label>
      <label>
        Notes
        <textarea
          value={notes}
          onChange={(event) => setNotes(event.target.value)}
          placeholder="Optional notes"
          rows={3}
        />
      </label>
      <button type="submit" disabled={!title.trim()}>
        Add task
      </button>
    </form>
  );
};
```

```tsx
import { Task } from "../types/task";
import { TaskItem } from "./TaskItem";

type TaskListProps = {
  tasks: Task[];
  onToggleDone: (taskId: string) => void;
};

export const TaskList = ({ tasks, onToggleDone }: TaskListProps) => {
  if (tasks.length === 0) {
    return <p className="empty-state">No tasks yet. Add one to get started.</p>;
  }

  return (
    <ul className="task-list">
      {tasks.map((task) => (
        <TaskItem key={task.id} task={task} onToggleDone={onToggleDone} />
      ))}
    </ul>
  );
};
```

`frontend/src/routes/TasksRoute.tsx`
```tsx
import { useState } from "react";
import { initialTasks } from "../data/mockTasks";
import { Task } from "../types/task";
import { TaskInput } from "../components/TaskInput";
import { TaskList } from "../components/TaskList";

const createMockTask = (title: string, notes: string): Task => ({
  id: `task-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
  title,
  notes: notes || undefined,
  completed: false,
});

export const TasksRoute = () => {
  const [tasks, setTasks] = useState(initialTasks);

  return (
    <section className="tasks-route">
      <h2>Task Center</h2>
      <TaskInput
        onAddTask={(title, notes) => {
          setTasks((prev) => [createMockTask(title, notes), ...prev]);
        }}
      />
      <TaskList
        tasks={tasks}
        onToggleDone={(taskId) => {
          setTasks((prev) =>
            prev.map((task) =>
              task.id === taskId ? { ...task, completed: !task.completed } : task
            )
          );
        }}
      />
    </section>
  );
};
```
