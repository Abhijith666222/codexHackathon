- `tasks-ui/state/task-state.ts`
```ts
import { NewTaskInput, Task } from "./task-model";

export interface TaskStateShape {
  schemaVersion: number;
  tasks: Task[];
}

export class InMemoryTaskState {
  private tasks = new Map<string, Task>();
  private nextId = 1;
  private nextTaskId() {
    return `task-${this.nextId++}`;
  }

  constructor(initial?: ReadonlyArray<Task>) {
    initial?.forEach((task) => {
      this.tasks.set(task.id, { ...task });
    });
  }

  createTask(input: NewTaskInput): Task {
    const now = Date.now();
    const task: Task = {
      id: this.nextTaskId(),
      title: input.title,
      description: input.description ?? null,
      isDone: false,
      createdAtMs: now,
      updatedAtMs: now,
    };
    this.tasks.set(task.id, task);
    return { ...task };
  }

  toggleTask(taskId: string): Task | null {
    const existing = this.tasks.get(taskId);
    if (!existing) {
      return null;
    }
    const updated: Task = {
      ...existing,
      isDone: !existing.isDone,
      updatedAtMs: Date.now(),
    };
    this.tasks.set(taskId, updated);
    return { ...updated };
  }

  removeTask(taskId: string): boolean {
    return this.tasks.delete(taskId);
  }

  listTasks(): Task[] {
    return Array.from(this.tasks.values()).map((task) => ({ ...task }));
  }

  getTask(taskId: string): Task | null {
    const task = this.tasks.get(taskId);
    return task ? { ...task } : null;
  }

  snapshot(): TaskStateShape {
    return {
      schemaVersion: 1,
      tasks: this.listTasks(),
    };
  }
}
```
